    def self.compile(resources=[])
      before_compile
      o = case resources
      when PoolParty::Cloud
        compile(resources.ordered_resources)
      when Array
        compile_array(resources)
      when PoolParty::Resource
        compile_resource(resources)
      end
      after_compile(o)
      o
    end
