  def append(other_hash)
    returning Hash.new do |h|
      h.merge!(self)

      end
    end
  end

  def method_missing(sym, *args, &block)
    if has_key?(sym)
      fetch(sym)
    elsif has_key?(sym.to_s)
      fetch(sym.to_s)
    else
      super
    end
  end
