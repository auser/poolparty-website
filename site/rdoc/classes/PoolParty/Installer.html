    def run
      begin
        self.__send__ :welcome_message
        steps.each {|c| self.__send__ c.to_sym }
        self.__send__ :closing_message
      rescue Exception => e
        colored_say e.inspect
        colored_say exit_msg
      ensure
        Colors.reset!
      end
    end

    def welcome_message
      welcome_msg = "We are going to take you through the installation process of PoolParty.\n\nFirst, we'll setup your environment so using PoolParty will be a <blue>breeze</blue>\n"

        @exit_msg = "\n<line>\n<yellow>Cancelled PoolParty installation</yellow>\n\nYou can always restart this by typing:\n<blue>cloud setup</blue>\n\n<line>\n"


      colored_say "<yellow>Welcome to PoolParty!</yellow>"
      colored_say welcome_msg
      colored_ask "Press enter to continue or Ctrl+C to exit", :no_value => true
      
    end
