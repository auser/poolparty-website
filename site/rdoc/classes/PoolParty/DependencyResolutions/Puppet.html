      def pretty_print_resources(pre=" ")
        returning Array.new do |out|
          resources.each do |name, res|
            out << "#{pre}#{name}"
            out << "#{pre*2}#{res.map {|a| a.name}}"
            res.each do |r|
              out << "#{pre*2}#{r.pretty_print_resources(pre*2)}"
            end
          end
        end.join("\n")
      end
