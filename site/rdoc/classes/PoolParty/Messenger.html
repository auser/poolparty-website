      with_socket(testing) do |sock|
        sock.send(msg, 0)

    def with_socket(testing=false, &block)
      host = testing ? "localhost" : (master.ip)
      vputs "Pinging #{host} with the messenger"
      socket = TCPSocket.open(host, Default.messenger_client_port)
      out = yield(socket)
      socket.close
      out
    end
