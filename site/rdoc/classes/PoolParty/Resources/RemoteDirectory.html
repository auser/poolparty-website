      def print_to_chef
        str = "remote_directory \"<%= name %>\" do\nsource <%= print_variable(source) %>\nfiles_backup <%= print_variable(files_backup) %>\nfiles_mode <%= print_variable(files_mode) %>\naction :<%= exists ? :create : :delete %>\nrecursive <%= print_variable(recursive) %>\nmode <%= print_variable(mode) %>\nowner <%= print_variable(owner) %>\ngroup <%= print_variable(group) %>\n"
        str << "  files_owner <%= print_variable(files_owner) %>\n" if files_owner
        str << "  files_group <%= print_variable(files_group) %>\n" if files_group
        str << "end"
      end
