      def print_to_chef
        str = "remote_file \"<%= name %>\" do\nsource <%= print_variable(source) %>\naction :<%= exists? ? :create : :delete %>\nbackup <%= backup %>\nmode <%= print_variable(mode) %>\nowner <%= print_variable(owner) %>\n"
        str << "  cookbook <%= print_variable(cookbook) %>\n" if cookbook
        str << "  checksum <%= print_variable(checksum) %>\n" if checksum
        str << "end"
      end
