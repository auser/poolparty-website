      def initialize(opts={}, prnt=nil, &block)
        setup_callbacks
        before_load(opts, &block)
        
        block = Proc.new {enable} unless block
        
        @opts = (opts.is_a?(Hash) ? opts : {:name => opts})
        super(opts, &block)
        
        run_in_context do
          loaded @opts, &block
        end
                
        after_create
      end
