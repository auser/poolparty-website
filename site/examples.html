Examples

This site is maintained using the following specfile 

   1  # Basic poolparty template
   2  require "poolparty_apache2_plugin"
   3  
   4  pool :cb do  
   5    instances 2..5
   6    port 80
   7    ami "ami-1cd73375" #Alestic's base Ubuntu AMI
   8    
   9    cloud :app do
  10          
  11      apache do
  12        installed_as_worker      
  13        
  14        has_virtualhost do        
  15          name "poolpartyrb.com"
  16          listen("8080")
  17          virtual_host_entry ::File.join(File.dirname(__FILE__), "cb/templates", "virtual_host.conf.erb")
  18          
  19          # We are going to have a repository that is updated across the servers
  20          has_git({:name => "poolpartyrepos", 
  21            :source => "git://github.com/auser/poolparty-website.git", 
  22            :cwd => "/var/www/poolpartyrb.com"}) do
  23          
  24            # We don't keep the site in the top level of the repos, so let's create
  25            # a symlink so that the public directory is a symlink of the root level
  26            # site directory
  27            has_symlink({:name => "/var/www/poolpartyrb.com/public", :from => "/var/www/poolpartyrb.com/repos/site"})
  28          end
  29        end
  30      end
  31      
  32    end
  33  
  34  end
