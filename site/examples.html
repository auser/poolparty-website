This site is maintained using the following specfile 

   1  # Basic poolparty template
   2  pool :cb do
   3    
   4    plugin_directory "#{File.dirname(__FILE__)}/plugins"
   5    instances 2..5
   6    port 80  
   7    # Alestic's base ubuntu AMI
   8    ami "ami-1cd73375"
   9    
  10    cloud :app do
  11      set_master_ip_to "75.101.162.232"
  12      mount_ebs_volume_at "vol-4e48ad27", "/vol"
  13      expand_when "cpu > 0.90"
  14      contract_when "cpu < 0.10"
  15      
  16      apache do
  17        installed_as_worker      
  18        
  19        has_virtualhost do |vh|
  20          name "poolpartyrb.com"
  21          listen("8080")
  22          virtual_host_entry ::File.join(File.dirname(__FILE__), "cb/templates", "virtual_host.conf.erb")
  23          
  24          # We are going to have a repository that is updated across the servers
  25          has_git({:name => "poolpartyrepos", 
  26            :source => "git://github.com/auser/poolparty-website.git", 
  27            :at => "/var/www/poolpartyrb.com"}) do |g|
  28              notify get_service("apache2")
  29              # We don't keep the site in the top level of the repos, so let's create
  30              # a symlink so that the public directory is a symlink of the root level
  31              # site directory
  32              # Requires is a temporary fix
  33              has_symlink({:name => "/var/www/poolpartyrb.com/public", :source => "/var/www/poolpartyrb.com/poolparty-website/site"})
  34              
  35            end
  36        end
  37      end
  38      
  39    end
  40  
  41  end
